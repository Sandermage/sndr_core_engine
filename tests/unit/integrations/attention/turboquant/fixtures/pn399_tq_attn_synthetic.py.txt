"""Synthetic TurboQuant attention backend stub for PN399 unit tests.

Contains the byte-exact anchor regions PN399 targets (sub-patches
A/B'/C/C2), arranged in a parse-valid file. NOT the real upstream file —
the live anchors are validated separately against a frozen live fixture.
Built from PN399's own anchor_old strings so the fixture cannot drift
out of sync with the patch.
"""
from __future__ import annotations

import torch


def is_workspace_manager_initialized():
    return False


def current_workspace_manager():
    return None


_genesis_pn353a_torch = torch


def _genesis_pn353a_round_up(a, b):
    return a


_CONTINUATION_DECODE_THRESHOLD = 64
_CONTINUATION_DECODE_MAX_CACHED_LEN = 32768

class TurboQuantImpl:
    def __init__(self, vllm_config):
        self.num_heads = 8
        self.head_size = 128
        self.max_num_kv_splits = (
            vllm_config.attention_config.tq_max_kv_splits_for_cuda_graph
        )

        # ════════════════════════════════════════════════════════════
        # [Genesis PN118 — backport of vllm#42551]
        # Pre-reserve decode scratch buffers so lock_workspace() at
        # end of warmup snapshots a workspace large enough for steady-
        # state decode. Without this, models whose warmup never lands
        # a decode forward through TQ (e.g. dense + hybrid attention,
        # 16/64 TQ layers in Lorbus 27B AutoRound) leave the workspace
        # locked at 0 MB and the first decode falls back to torch.empty
        # every layer/forward.
        # ════════════════════════════════════════════════════════════
        self._reserve_decode_workspace(vllm_config)

    def _reserve_decode_workspace(self, vllm_config) -> None:
        if not is_workspace_manager_initialized():
            return
        manager = current_workspace_manager()
        if manager.is_locked():
            return
        if not hasattr(manager, 'reserve'):
            # PN118 workspace-side patch did not apply. Skip silently.
            return
        scheduler_config = vllm_config.scheduler_config
        speculative_config = vllm_config.speculative_config
        extra_spec_tokens = (
            speculative_config.num_speculative_tokens
            if speculative_config is not None else 0
        )
        max_batch_tokens = scheduler_config.max_num_seqs * (1 + extra_spec_tokens)
        query_dtype = vllm_config.model_config.dtype
        manager.reserve(
            (
                (
                    max_batch_tokens,
                    self.num_heads,
                    self.max_num_kv_splits,
                    self.head_size + 1,
                ),
                torch.float32,
            ),
            ((max_batch_tokens, self.num_heads, self.head_size), query_dtype),
            ((max_batch_tokens, self.num_heads), torch.float32),
        )

    def _decode_attention(self, query):
        B = query.shape[0]
        D = self.head_size
        S = self.max_num_kv_splits
        Hq = self.num_heads
        mid_o_buf = output_buf = lse_buf = None
        # ════════════════════════════════════════════════════════════
        # [Genesis PN118 — backport of vllm#42551]
        # Use try_get_simultaneous: returns None if workspace is locked
        # and undersized (instead of raising AssertionError). Caller
        # falls back to torch.empty so the engine keeps serving.
        # ════════════════════════════════════════════════════════════
        if is_workspace_manager_initialized():
            manager = current_workspace_manager()
            if hasattr(manager, 'try_get_simultaneous'):
                bufs = manager.try_get_simultaneous(
                    ((B, Hq, S, D + 1), torch.float32),
                    ((B, Hq, D), query.dtype),
                    ((B, Hq), torch.float32),
                )
                if bufs is not None:
                    mid_o_buf, output_buf, lse_buf = bufs
            else:
                # PN118 workspace-side not applied — use legacy path.
                mid_o_buf, output_buf, lse_buf = manager.get_simultaneous(
                    ((B, Hq, S, D + 1), torch.float32),
                    ((B, Hq, D), query.dtype),
                    ((B, Hq), torch.float32),
                )
        if mid_o_buf is None:
            mid_o_buf = torch.empty(
                (B, Hq, S, D + 1), dtype=torch.float32, device=query.device
            )
            output_buf = torch.empty((B, Hq, D), dtype=query.dtype, device=query.device)
            lse_buf = torch.empty((B, Hq), dtype=torch.float32, device=query.device)
        return mid_o_buf, output_buf, lse_buf


class TurboQuantMetadataBuilder:
    def __init__(self):
        self.vllm_config = None
        self.kv_cache_spec = None

    def _reserve_workspace(self):
        model_config = self.vllm_config.model_config
        scheduler_config = self.vllm_config.scheduler_config
        max_num_reqs = scheduler_config.max_num_seqs
        num_heads = 8
        num_kv_heads = 8
        head_size = 128
        try:
            max_num_splits = (
                self.vllm_config.attention_config.tq_max_kv_splits_for_cuda_graph
            )
        except AttributeError:
            return

        # Decode scratch — mirrors _decode_attention's get_simultaneous.
        current_workspace_manager().get_simultaneous(
            (
                (max_num_reqs, num_heads, max_num_splits, head_size + 1),
                _genesis_pn353a_torch.float32,
            ),
            ((max_num_reqs, num_heads, head_size), model_config.dtype),
            ((max_num_reqs, num_heads), _genesis_pn353a_torch.float32),
        )

        # Continuation-prefill K/V dequant buffers — only when chunked
        # prefill is enabled and batch is large enough to actually use
        # the continuation path (gated by _CONTINUATION_DECODE_THRESHOLD).
        reserve_cont = (
            scheduler_config.enable_chunked_prefill
            and scheduler_config.max_num_batched_tokens
            > _CONTINUATION_DECODE_THRESHOLD
        )
        if not reserve_cont:
            return
        max_cached_len = max(0, model_config.max_model_len - 1)
        alloc_len = _genesis_pn353a_round_up(
            max_cached_len, self.kv_cache_spec.block_size
        )
        cache_buf_shape = (1, num_kv_heads, alloc_len, head_size)
        current_workspace_manager().get_simultaneous(
            (cache_buf_shape, _genesis_pn353a_torch.float16),
            (cache_buf_shape, _genesis_pn353a_torch.float16),
        )
